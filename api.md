### getNewsPage

```
POST

data:{
  page:int，
  type:'string' // [u'所有',u'财经',u'教育',u'科技',u'社会',u'时尚',u'时政',u'体育']
}
```

响应

```
data:{
  errorCode:int
  may errorMsg:string
  data:{
    [
    type:"news",
    title:string,
    url:string,
    hot:int,
    <!-- 标签 -->
    label:string,
    keywords:[string,string...],
    <!-- 摘要 -->
    abstract:string,
    fromTopic:string,
    ],
    [
    type:"group",
    title:string,
    relatedNews:[{
      title:'',
      url:''
      }],
    hot:int,
    history:[int,int...],
    keywords:[string,...]
    ]
    }
}
```
